### ✅ 2. **Fix Your Script Tag**

Update your `<script>` section at the bottom to include this:

```js
document.addEventListener("DOMContentLoaded", () => {
  const savedTheme = localStorage.getItem("theme") || "dark";
  document.documentElement.setAttribute("data-theme", savedTheme);

  // ✅ Properly initialize Lucide icons after DOM is fully loaded
  lucide.createIcons();

  // ✅ OPTIONAL: Protect hero content from being changed (if needed)
  const hero = document.querySelector('.hero-content');
  const observer = new MutationObserver(() => {
    lucide.createIcons(); // re-render if anything changes
  });
  if (hero) {
    observer.observe(hero, { childList: true, subtree: true });
  }

  // OPTIONAL: renderTemplates(); ← Only if you're using it
});
```
